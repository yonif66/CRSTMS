========================================================================
                     CRSTMS DATABASE SETUP GUIDE
========================================================================

Model Title:      Relational Database Engine Schema for CRSTMS
Database Type:    MySQL / MariaDB
Name:             crstms_db
Character Set:    utf8mb4
Collation:        utf8mb4_unicode_ci

This guide defines the table structures, foreign key constraints,
example seeding queries, and validation procedures required to populate
your Database Engine for academic defense.

------------------------------------------------------------------------
PART 1: CREATING DATABASE
------------------------------------------------------------------------
To create your database manually using command-line SQL client or phpMyAdmin:

```sql
CREATE DATABASE IF NOT EXISTS crstms_db
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE crstms_db;
```

------------------------------------------------------------------------
PART 2: SQL SCHEMA CREATION
------------------------------------------------------------------------
Execute the following statements to structure the relational table schemas:

```sql
-- 1. users Table (Staff roles & Login security)
CREATE TABLE IF NOT EXISTS `users` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `username` VARCHAR(50) NOT NULL UNIQUE,
  `email` VARCHAR(100) NOT NULL UNIQUE,
  `password_hash` VARCHAR(255) NOT NULL,
  `full_name` VARCHAR(100) NOT NULL,
  `role` ENUM('Admin', 'Receptionist', 'Technician', 'Delivery', 'Customer') NOT NULL,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- 2. customers Table (CRM files)
CREATE TABLE IF NOT EXISTS `customers` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `user_id` INT NULL,
  `full_name` VARCHAR(100) NOT NULL,
  `email` VARCHAR(100) NOT NULL UNIQUE,
  `phone` VARCHAR(20) NOT NULL,
  `address` VARCHAR(255) NOT NULL,
  `alt_phone` VARCHAR(20) NULL,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE SET NULL
) ENGINE=InnoDB;

-- 3. devices Table (Customer Hardware parameters)
CREATE TABLE IF NOT EXISTS `devices` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `customer_id` INT NOT NULL,
  `device_type` VARCHAR(50) NOT NULL,
  `brand` VARCHAR(50) NOT NULL,
  `model` VARCHAR(50) NOT NULL,
  `serial_number` VARCHAR(100) NOT NULL UNIQUE,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`customer_id`) REFERENCES `customers`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB;

-- 4. repair_tickets Table (Primary Trackers)
CREATE TABLE IF NOT EXISTS `repair_tickets` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `device_id` INT NOT NULL,
  `customer_id` INT NOT NULL,
  `issue_description` TEXT NOT NULL,
  `status` ENUM('Pending Diagnosis', 'In Progress', 'Awaiting Parts', 'Completed', 'Handed Over') NOT NULL DEFAULT 'Pending Diagnosis',
  `assigned_technician` INT NULL,
  `technician_name` VARCHAR(100) NULL,
  `work_order_notes` TEXT NULL,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`device_id`) REFERENCES `devices`(`id`) ON DELETE CASCADE,
  FOREIGN KEY (`customer_id`) REFERENCES `customers`(`id`) ON DELETE CASCADE,
  FOREIGN KEY (`assigned_technician`) REFERENCES `users`(`id`) ON DELETE SET NULL
) ENGINE=InnoDB;

-- 5. diagnostics Table (Diagnosis checklists)
CREATE TABLE IF NOT EXISTS `diagnostics` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `ticket_id` INT NOT NULL,
  `technician_id` INT NOT NULL,
  `defect_found` TEXT NOT NULL,
  `diagnostic_notes` TEXT NULL,
  `estimated_repair_cost` DECIMAL(10,2) NOT NULL,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`ticket_id`) REFERENCES `repair_tickets`(`id`) ON DELETE CASCADE,
  FOREIGN KEY (`technician_id`) REFERENCES `users`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB;

-- 6. inventory Table (Spare component catalog)
CREATE TABLE IF NOT EXISTS `inventory` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `part_name` VARCHAR(100) NOT NULL,
  `serial_number` VARCHAR(100) NOT NULL UNIQUE,
  `stock_quantity` INT NOT NULL DEFAULT 0,
  `unit_price` DECIMAL(10,2) NOT NULL,
  `low_stock_threshold` INT NOT NULL DEFAULT 2,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

-- 7. invoices Table (Billing ledgers with Ethiopian 15% VAT)
CREATE TABLE IF NOT EXISTS `invoices` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `ticket_id` INT NOT NULL,
  `customer_id` INT NOT NULL,
  `subtotal` DECIMAL(10,2) NOT NULL,
  `vat_amount` DECIMAL(10,2) NOT NULL, -- Calculated at 15% Standard Ethiopian VAT rate
  `grand_total` DECIMAL(10,2) NOT NULL,
  `payment_status` ENUM('Unpaid', 'Pending Review', 'Paid') NOT NULL DEFAULT 'Unpaid',
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`ticket_id`) REFERENCES `repair_tickets`(`id`) ON DELETE CASCADE,
  FOREIGN KEY (`customer_id`) REFERENCES `customers`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB;

-- 8. receipt_uploads Table (Payment transfer check slips)
CREATE TABLE IF NOT EXISTS `receipt_uploads` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `invoice_id` INT NOT NULL,
  `uploaded_filename` VARCHAR(255) NOT NULL,
  `uploaded_image_url` VARCHAR(255) NOT NULL,
  `status` ENUM('Pending', 'Approved', 'Rejected') NOT NULL DEFAULT 'Pending',
  `admin_notes` TEXT NULL,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`invoice_id`) REFERENCES `invoices`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB;

-- 9. deliveries Table (Courier Dispatch logs)
CREATE TABLE IF NOT EXISTS `deliveries` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `ticket_id` INT NOT NULL,
  `destination_address` VARCHAR(255) NOT NULL,
  `assigned_courier` VARCHAR(100) NOT NULL,
  `delivery_type` ENUM('Pickup', 'Dropoff') NOT NULL,
  `status` ENUM('Pending', 'Dispatched', 'Handed Over') NOT NULL DEFAULT 'Pending',
  `notes` TEXT NULL,
  `updated_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`ticket_id`) REFERENCES `repair_tickets`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB;

-- 10. audit_logs Table (Operational accountability logging)
CREATE TABLE IF NOT EXISTS `audit_logs` (
  `id` INT AUTO_INCREMENT PRIMARY KEY,
  `user_id` INT NOT NULL DEFAULT 0,
  `username` VARCHAR(100) NOT NULL,
  `action_type` VARCHAR(50) NOT NULL,
  `affected_module` VARCHAR(100) NOT NULL,
  `details` TEXT NOT NULL,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;
```

PART 4: DATABASE VERIFICATION INSTRUCTIONS
------------------------------------------------------------------------
To verify configuration logs are reading relationships correctly:

1. Display all imported tables:
   ```sql
   SHOW TABLES;
   ```
2. Verify pre-loaded accounts list:
   ```sql
   SELECT COUNT(*) FROM users;
   ```
3. Check low inventory items:
   ```sql
   SELECT * FROM inventory WHERE stock_quantity <= low_stock_threshold;
   ```

PART 5: BACKUP & RESTORE COMMANDS
------------------------------------------------------------------------
To export db for testing copies:
```bash
mysqldump -u root -p crstms_db > crstms_backup.sql
```

To restore database schema:
```bash
mysql -u root -p crstms_db < crstms_backup.sql
```
